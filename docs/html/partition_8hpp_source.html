119  static vector<vector<T>>* generate(vector<T> pivots, vector<T> elements)
120  /*
121  * Generate a partition of the elements vector with the supplied pivots
122  * NOTE: pivots will be sorted
123  */
124  {
125  unsigned int npivots = pivots.size();
126  vector<vector<T>>* parts = new vector<vector<T>>(npivots+1, vector<T>());
127 
128  for(T element: elements)
129  {
130  if (element < pivots[0])
131  parts->at(0).push_back(element);
132  else if (pivots[npivots-1] < element)
133  parts->at(npivots).push_back(element);
134  else
135  {
136  for(int k=0; k<npivots-1; k++)
137  {
138  if(pivots[k] < element && element < pivots[k+1])
139  parts->at(k+1).push_back(element);
140  }
141  }
142  }
143  return parts;
144  }
