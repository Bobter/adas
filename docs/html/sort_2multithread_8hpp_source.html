45  template <class RandomAccessIterator>
46  void multi_merge(vector<Boundaries<RandomAccessIterator>> limits, bool verbose)
47  /*
48  * Merge several sorted parts of a array into it.
49  * (using an auxiliary vector to stored intermediate merges)
50  */
51  {
52  unsigned int size = limits.size();
53  unsigned int nelems = 0;
54  for(Boundaries<RandomAccessIterator> limit : limits)
55  nelems += distance(limit.init, limit.end);
56 
57  // auxiliary vector to store intermediate merges
58  vector<typename RandomAccessIterator::value_type> merge_aux;
59  merge_aux = vector<typename RandomAccessIterator::value_type>(nelems);
60  RandomAccessIterator merge_itr = merge_aux.begin();
61  RandomAccessIterator init_merge = merge_aux.begin();
62 
63  Boundaries<RandomAccessIterator> limit = limits[0];
64  RandomAccessIterator init = limit.init;
65 
66  // merge sorted sub containers
67  for(unsigned int k=1; k<size; k++)
68  {
69  Boundaries<RandomAccessIterator> next_limit = limits[k];
70  merge_itr = serial::merge(limit.init, limit.end,
71  next_limit.init, next_limit.end,
72  merge_itr);
73  copy(init_merge, merge_itr, init); // overwrite partitions by merged elements
74  limit.end = next_limit.end;
75  limit.init = init; // limit.init iterator is modify (incremented) in serial::merge
76  merge_itr = init_merge;
77  if(verbose)
78  print::to_stdout("merge:", merge_aux); // ONLY FOR DEBUGING PURPOSES
79  }
80  }

83  template<class T>
84  vector<T> quicksort(vector<T> elements, vector<T> pivots, bool verbose)
85  /*
86  * Improve efficiency of quicksort algorithm:
87  * + Reimplement Partition<T> class using pointer (to avoid copy data)
88  * + Parallelize for loop in quicksort body
89  * + change of sortng algorithm to sort partition's parts (codify mergesort algorithm)
90  */
91  {
92  Partition<T> partition = Partition<T>(pivots, elements);
93  unsigned size = partition.size();
94 
95 #pragma omp parallel for shared(partition) firstprivate(size)
96  for(unsigned int k = 0; k < size; k++)
97  {
98  vector<T> part = partition.get_part(k);
99  if(1 < part.size()) // if part has some elements, otherwise it's sorted
100  {
101  serial::bubble(part.begin(), part.end(), verbose);
102  // ONLY FOR DEBUG PURPOSE
103 #pragma omp critical
104  {
105  unsigned int idthread = omp_get_thread_num();
106  cout << "Thread " << idthread << ": ";
107  print::to_stdout(part);
108  }
109  partition.set_part(k, part);
110  }
111  }
112 
113  return partition.join();
114  }

117  template<class RandomAccessIterator, class SerialSortAlgorithm>
118  void spm(RandomAccessIterator first, RandomAccessIterator last,
119  unsigned int omp_nthreads, SerialSortAlgorithm ssa, bool verbose)
120  /*
121  * spm : split, process and merge
122  * This algorithm sort a container in parallel using some serial sort algorithm.
123  * Splitting it by the number of threads, then proces each part in parallel with a serial algorithm
124  * and finaly merge the sorted parts.
125  */
126  {
127  unsigned int d = distance(first, last);
128  unsigned int nthreads = omp_nthreads;
129  omp_set_num_threads(nthreads);
130  vector<Boundaries<RandomAccessIterator>> limits(nthreads);
131 
132 #pragma omp parallel shared(limits) firstprivate(first, last, verbose, nthreads)
133  {
134  unsigned int idthread = omp_get_thread_num();
135  RandomAccessIterator init = first + idthread*(d/nthreads);
136  RandomAccessIterator end = first + (idthread+1)*(d/nthreads);
137 
138  if(idthread == nthreads-1)
139  end = last;
140 
141  Boundaries<RandomAccessIterator> limit;
142  limit.init = init;
143  limit.end = end;
144 
145  limits[idthread] = limit;
146 
147  ssa(init, end, false);
148  //ONLY FOR DEBUGING PURPOSES
149  if(verbose)
150  {
151 #pragma omp critical
152  {
153  print::to_stdout("Thread " + to_string(idthread) + ":", init, end);
154  }
155  }
156  }
157 
158  multi_merge(limits, verbose);
159  }
