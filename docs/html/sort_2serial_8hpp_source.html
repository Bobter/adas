156  void insertion(BidirectionalIterator first, BidirectionalIterator last, bool verbose)
157  /*
158  * Supported containers: vector<T>, array[T]
159  */
160  {
161  BidirectionalIterator prev2first = prev(first);
162  BidirectionalIterator itr;
163  BidirectionalIterator inner;
164 
165  for(itr=first+1; itr != last; itr++)
166  {
167  auto key = *itr;
168  inner = prev(itr);
169  while(inner != prev2first && key < *inner)
170  {
171  *next(inner) = *inner;
172  inner--;
173  }
174  *next(inner) = key;
175 
176  if (verbose)
177  au::print::to_stdout(first, last);
178  }

252 
253  for(it=init; it != last; it++)
254  counter[*it] += 1;
255 
256  if(verbose)
257  au::print::to_stdout("(init) counter:", counter);
258 
259  for(it=counter.begin(); it != counter.end(); it++)
260  *it += *prev(it);
261 
262  if(verbose)
263  au::print::to_stdout("(accumulated) counter:", counter);
264 
265  for(it=prev(last); it != prev(init); it--)
266  {
267  sorted[counter[*it]-1] = *it;
268  counter[*it] -= 1;
269  if(verbose)

271  au::print::to_stdout("sorted:", sorted);
